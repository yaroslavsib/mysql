CREATE DATABASE photo_;

CREATE TABLE Users (user_id int not null primary key autoincrement,
name varchar(20) not null);

CREATE TABLE Photo (photo_id int not null,
photo_name, varchar(20) not null,
id_own int not null,
status varchar(20) not null);

CREATE TABLE Album (Album_id int not null,
album_name, varchar(20) not null,
id_photo int not null,
id_own int not null);

CREATE TABLE Friendship (User_id int not null,
friend_id int not null);
